    const height = 400;                 //fönster höjd
    const width = 400;

    const dataArr = [75, 80, 85, 90, 95, 100];
    
    const canvas = d3.select("#canvas") //skapa objekt
    
    canvas.append("svg")                //addera svg till canvas
        .attr('height', height)         //lägg till attribut
        .attr('width', width);

    const svg = canvas.select('svg');   //skapa svg objekt
    const rect = svg.selectAll('rect'); //skapa rect genom att välja ut 
                                        //alla rect (som ännu inte finns)

    rect.data(dataArr)                  //med alla datavärden i array, välj rect
        .enter()                        //nyckeln till välja rect 
                                        //som ännu inte finns
        .append("rect")
        .attr('height', height)         
        .attr('fill', function(){       //fyll med slumpfärger
            let color = Math.floor(Math.random() * Math.pow(256, 3)).toString(16);
            while(color.length < 6) {
                color = "0" + color;
            }
            return "#" + color;
        })

        .attr('width', function(d, i){  
            return (width / dataArr.length - 1);
        })

        .attr('x', function(d, i){
            let myWidth = i * (width / dataArr.length)
            return myWidth;
        })

        .attr('y', function(d, i) {
            console.log(d3.max(dataArr));
            return height - (d*2)
        });
